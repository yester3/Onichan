import random
import aiohttp
import asyncio

class UsernameChecker:
    def __init__(self, tokens):
        self.tokens = tokens
        self.current_token_index = 0
        self.running = False
        self.session = None
        
        # Configuración de símbolos
        self.symbols = {
            'digits': '0123456789',
            'letters': 'abcdefghijklmnopqrstuvwxyz',
            'underlines': '_'
        }
        
        # Rate limit tracking
        self.rate_limited_until = {}
        self.consecutive_errors = 0
        
    def get_current_token(self):
        """Obtiene el token actual"""
        if not self.tokens:
            return None
        return self.tokens[self.current_token_index]
    
    def rotate_token(self):
        """Rota al siguiente token (entre las 3 cuentas)"""
        if len(self.tokens) <= 1:
            return
        
        self.current_token_index = (self.current_token_index + 1) % len(self.tokens)
        print(f"[TOKEN] Rotado a cuenta {self.current_token_index + 1}/3")
    
    def generate_username(self, length=None, only_3_4=False):
        """Genera un username aleatorio válido para Discord"""
        if only_3_4:
            length = random.choice([3, 4])
        elif length is None:
            length = random.randint(2, 4)
        
        chars = ''
        chars += self.symbols['letters']
        chars += self.symbols['digits']
        chars += self.symbols['underlines']
        
        # Username debe empezar con letra
        username = random.choice(self.symbols['letters'])
        
        # Resto del username
        for _ in range(length - 1):
            username += random.choice(chars)
        
        return username
    
    async def check_username(self, username):
        """
        Verifica si un username está disponible usando la API de Discord
        Retorna: dict con 'available', 'error', 'retry_after'
        """
        token = self.get_current_token()
        if not token:
            return {'available': False, 'error': True, 'retry_after': 5, 'message': 'No tokens available'}
        
        # Verificar si el token está rate limited
        if token in self.rate_limited_until:
            if asyncio.get_event_loop().time() < self.rate_limited_until[token]:
                print(f"[RATE LIMIT] Cuenta {self.current_token_index + 1} limitada, rotando...")
                self.rotate_token()
                token = self.get_current_token()
        
        headers = {
            'Authorization': token,
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36',
            'Accept': '*/*',
            'Accept-Language': 'en-US,en;q=0.9',
            'Content-Type': 'application/json',
            'Origin': 'https://discord.com',
            'Referer': 'https://discord.com/channels/@me',
            'X-Super-Properties': 'eyJvcyI6IldpbmRvd3MiLCJicm93c2VyIjoiQ2hyb21lIiwiZGV2aWNlIjoiIiwic3lzdGVtX2xvY2FsZSI6ImVuLVVTIiwiYnJvd3Nlcl91c2VyX2FnZW50IjoiTW96aWxsYS81LjAgKFdpbmRvd3MgTlQgMTAuMDsgV2luNjQ7IHg2NCkgQXBwbGVXZWJLaXQvNTM3LjM2IChLSFRNTCwgbGlrZSBHZWNrbykgQ2hyb21lLzEyMC4wLjAuMCBTYWZhcmkvNTM3LjM2IiwiYnJvd3Nlcl92ZXJzaW9uIjoiMTIwLjAuMC4wIiwib3NfdmVyc2lvbiI6IjEwIiwicmVmZXJyZXIiOiJodHRwczovL2Rpc2NvcmQuY29tL2NoYW5uZWxzL0BtZSIsInJlZmVycmluZ19kb21haW4iOiJkaXNjb3JkLmNvbSIsInJlZmVycmVyX2N1cnJlbnQiOiIiLCJyZWZlcnJpbmdfZG9tYWluX2N1cnJlbnQiOiIiLCJyZWxlYXNlX2NoYW5uZWwiOiJzdGFibGUiLCJjbGllbnRfYnVpbGRfbnVtYmVyIjoyNTg0NzAsImNsaWVudF9ldmVudF9zb3VyY2UiOm51bGwsImRlc2lnbl9pZCI6MH0=',
            'X-Discord-Locale': 'en-US',
            'X-Discord-Timezone': 'America/New_York',
        }
        
        # Payload para verificar disponibilidad
        payload = {
            "username": username
        }
        
        try:
            async with self.session.post(
                'https://discord.com/api/v9/users/@me/pomelo',
                headers=headers,
                json=payload,
                timeout=aiohttp.ClientTimeout(total=10)
            ) as response:
                
                status = response.status
                response_data = await response.text()
                
                print(f"[API] {username} -> Status {status}")
                
                # 200 = Cambiado exitosamente (disponible)
                if status == 200:
                    print(f"[✅] {username} DISPONIBLE")
                    self.consecutive_errors = 0
                    return {'available': True, 'error': False, 'retry_after': 2}
                
                # 400 = Error de validación (probablemente no disponible)
                elif status == 400:
                    self.consecutive_errors = 0
                    if 'username' in response_data.lower():
                        print(f"[❌] {username} NO disponible")
                        return {'available': False, 'error': False, 'retry_after': 1.5}
                    return {'available': False, 'error': True, 'retry_after': 2}
                
                # 429 = Rate limited
                elif status == 429:
                    retry_after = 60
                    try:
                        json_data = await response.json()
                        retry_after = json_data.get('retry_after', 60)
                    except:
                        pass
                    
                    print(f"[⚠️] Rate limit en cuenta {self.current_token_index + 1}. Esperar {retry_after}s")
                    self.rate_limited_until[token] = asyncio.get_event_loop().time() + retry_after
                    self.rotate_token()
                    
                    return {'available': False, 'error': True, 'retry_after': min(retry_after, 5)}
                
                # 401 = Token inválido
                elif status == 401:
                    print(f"[❌] Cuenta {self.current_token_index + 1} inválida, rotando...")
                    self.consecutive_errors += 1
                    self.rotate_token()
                    return {'available': False, 'error': True, 'retry_after': 1}
                
                # Otros errores
                else:
                    self.consecutive_errors += 1
                    print(f"[⚠️] Error {status}: {response_data[:100]}")
                    return {'available': False, 'error': True, 'retry_after': 2}
                    
        except aiohttp.ClientError as e:
            print(f"[ERROR] Client error: {e}")
            self.consecutive_errors += 1
            return {'available': False, 'error': True, 'retry_after': 3}
        except asyncio.TimeoutError:
            print(f"[ERROR] Timeout checking {username}")
            return {'available': False, 'error': True, 'retry_after': 2}
        except Exception as e:
            print(f"[ERROR] Exception: {e}")
            self.consecutive_errors += 1
            return {'available': False, 'error': True, 'retry_after': 2}
    
    async def close(self):
        """Cierra la sesión HTTP"""
        if self.session:
            await self.session.close()
